      )}

 

      <div style={{position:"fixed",bottom:0,left:"50%",transform:"translateX(-50%)",width:"100%",maxWidth:480,background:"rgba(15,17,23,0.96)",backdropFilter:"blur(12px)",borderTop:`1px solid ${BORDER}`,display:"flex",justifyContent:"space-around",padding:"8px 0",paddingBottom:"max(10px,env(safe-area-inset-bottom))"}}>

        {TABS.map(([id,icon,label])=>(

          <button key={id} onClick={()=>setTab(id)} style={{display:"flex",flexDirection:"column",alignItems:"center",gap:2,background:"none",border:"none",cursor:"pointer",color:tab===id?ACCENT:MUTED,padding:"4px 16px",fontFamily:"Inter,sans-serif",transition:"color 0.2s"}}>

            <span style={{fontSize:"1.15rem"}}>{icon}</span>

      </div>

    </div>

  );
